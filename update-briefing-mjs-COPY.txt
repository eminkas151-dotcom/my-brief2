function google(query) {
  return GOOGLE_BASE + encodeURIComponent(query);
}

function decodeEntities(text = '') {
  return String(text)
    .replace(/<!\[CDATA\[/g, '')
    .replace(/\]\]>/g, '')
    .replace(/&nbsp;/g, ' ')
    .replace(/&amp;/g, '&')
    .replace(/&quot;/g, '"')
    .replace(/&#39;/g, "'")
    .replace(/&apos;/g, "'")
    .replace(/&lt;/g, '<')
    .replace(/&gt;/g, '>')
    .replace(/&#(\d+);/g, (_, n) => String.fromCharCode(Number(n)))
    .replace(/&#x([0-9a-f]+);/gi, (_, n) => String.fromCharCode(parseInt(n, 16)));
}

function strip(html = '') {
  return decodeEntities(String(html))
    .replace(/<script[\s\S]*?<\/script>/gi, ' ')
    .replace(/<style[\s\S]*?<\/style>/gi, ' ')
    .replace(/<[^>]+>/g, ' ')
    .replace(/\s+/g, ' ')
    .trim();
}

function getTag(block, name) {
  const re = new RegExp(`<${name}(?:\\s[^>]*)?>([\\s\\S]*?)<\\/${name}>`, 'i');
  const m = block.match(re);
  return m ? strip(m[1]) : '';
}

function getLink(block) {
  const normal = getTag(block, 'link');
  if (normal) return normal;
  const atom = block.match(/<link[^>]+href=["']([^"']+)["']/i);
  return atom ? decodeEntities(atom[1]) : '';
}

function getDate(block) {
  return getTag(block, 'pubDate') || getTag(block, 'published') || getTag(block, 'updated') || getTag(block, 'dc:date') || '';
}

function parseFeed(xml, source) {
  const blocks = [...String(xml).matchAll(/<item[\s\S]*?<\/item>|<entry[\s\S]*?<\/entry>/gi)].map(x => x[0]);
  return blocks.map(block => {
    let title = getTag(block, 'title');
    let text = getTag(block, 'description') || getTag(block, 'summary') || getTag(block, 'content:encoded') || getTag(block, 'content');
    let url = getLink(block);
    let publishedAt = getDate(block);

    // Google News often appends source to the title. Keep source readable.
    if (source === 'Google News' && title.includes(' - ')) {
      const parts = title.split(' - ');
      const possibleSource = parts.pop();
      title = parts.join(' - ').trim() || title;
      source = possibleSource || source;
    }

    return { title, text, url, publishedAt, source };
  }).filter(x => x.title && x.url);
}

async function fetchFeed([source, url]) {
  const controller = new AbortController();
  const timer = setTimeout(() => controller.abort(), 20000);
  try {
    const res = await fetch(url, {
      signal: controller.signal,
      headers: { 'user-agent': UA, 'accept': 'application/rss+xml, application/xml, text/xml, */*' }
    });
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    const xml = await res.text();
    const items = parseFeed(xml, source);
    console.log(`✓ ${source}: ${items.length}`);
    return items;
  } catch (e) {
    console.log(`× ${source}: ${e.message}`);
    return [];
  } finally {
    clearTimeout(timer);
  }
}

function normalizeKey(title) {
  return String(title || '')
    .toLowerCase()
    .replace(/[^а-яёa-z0-9]+/gi, ' ')
    .replace(/\b(в|на|и|о|об|по|для|с|со|the|a|an|of|to|in|on)\b/g, ' ')
    .replace(/\s+/g, ' ')
    .trim()
    .slice(0, 120);
}

function enrich(item, sectionTitle) {
  let text = item.text || '';
  text = text.replace(/^Источник:\s*/i, '').trim();

  if (text.length < 120) {
    text = `${item.title}. Материал опубликован источником «${item.source}» и добавлен в раздел «${sectionTitle.replace(/^\S+\s*/, '')}». Подробности, контекст и возможные обновления лучше смотреть в полной версии по ссылке на источник.`;
  }

  if (text.length > 650) text = text.slice(0, 647).trim() + '…';

  let dateValue = Date.parse(item.publishedAt || '');
  if (Number.isNaN(dateValue)) dateValue = Date.now();

  return {
    title: item.title.trim(),
    text,
    source: item.source,
    url: item.url,
    publishedAt: new Date(dateValue).toISOString()
  };
}

function unique(items) {
  const seen = new Set();
  const result = [];
  for (const item of items) {
    const key = normalizeKey(item.title);
    if (!key || seen.has(key)) continue;
    seen.add(key);
    result.push(item);
  }
  return result;
}

async function buildSection(section) {
  const raw = (await Promise.all(section.feeds.map(fetchFeed))).flat();
  const items = unique(raw)
    .map(item => enrich(item, section.title))
    .sort((a, b) => Date.parse(b.publishedAt) - Date.parse(a.publishedAt))
    .slice(0, section.limit);

  console.log(`${section.title}: ${items.length} новостей`);

  if (!items.length) {
    items.push({
      title: section.title.replace(/^\S+\s*/, ''),
      text: 'В этом разделе пока не удалось получить свежие материалы из RSS. Следующий запуск попробует снова. Сайт и автоматическое обновление при этом работают.',
      source: 'My Brief',
      url: '',
      publishedAt: new Date().toISOString()
    });
  }

  return { id: section.id, title: section.title, items };
}

const sections = await Promise.all(SECTIONS.map(buildSection));

  sections
};
