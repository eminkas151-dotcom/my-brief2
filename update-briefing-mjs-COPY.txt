function strip(html=''){return String(html).replace(/<!\[CDATA\[/g,'').replace(/\]\]>/g,'').replace(/<[^>]+>/g,' ').replace(/&nbsp;/g,' ').replace(/&amp;/g,'&').replace(/&quot;/g,'"').replace(/&#39;/g,"'").replace(/\s+/g,' ').trim()}
function tag(block,name){const m=block.match(new RegExp(`<${name}[^>]*>([\\s\\S]*?)<\\/${name}>`,'i'));return m?strip(m[1]):''}
function link(block){const l=tag(block,'link');if(l)return l;const m=block.match(/<link[^>]+href=["']([^"']+)/i);return m?m[1]:''}
function parse(xml,source){const blocks=[...xml.matchAll(/<item[\s\S]*?<\/item>|<entry[\s\S]*?<\/entry>/gi)].map(x=>x[0]);return blocks.map(b=>({title:tag(b,'title'),text:tag(b,'description')||tag(b,'summary')||tag(b,'content:encoded'),url:link(b),publishedAt:tag(b,'pubDate')||tag(b,'published')||tag(b,'updated'),source})).filter(x=>x.title)}
function enrich(item){let text=item.text||'';if(text.length<110){text = `${item.title}. ${text || 'Подробности уточняются.'} Материал добавлен в ленту, чтобы быстро видеть главные события по разделу. Для полной версии откройте источник.`}if(text.length>520) text=text.slice(0,517)+'…';return {...item,text}}
async function fetchFeed([source,url]){try{const controller=new AbortController();const t=setTimeout(()=>controller.abort(),15000);const r=await fetch(url,{signal:controller.signal,headers:{'user-agent':'MyBriefBot/1.0'}});clearTimeout(t);if(!r.ok) throw new Error(`${r.status}`);return parse(await r.text(),source)}catch(e){console.warn(`Не удалось загрузить ${source}: ${url} (${e.message})`);return []}}
function unique(items){const seen=new Set();return items.filter(i=>{const key=(i.title||'').toLowerCase().replace(/[^а-яa-z0-9]+/gi,' ').trim();if(!key||seen.has(key))return false;seen.add(key);return true})}
async function sectionData(s){const results=(await Promise.all(s.feeds.map(fetchFeed))).flat();let items=unique(results).map(enrich).slice(0,s.limit);if(!items.length){items=[{title:s.title.replace(/^[^\s]+\s*/,''),text:fallbackText[s.id],source:'My Brief',url:''}]};return {id:s.id,title:s.title,items}}
const briefing={updatedAt:new Date().toISOString(),sections:await Promise.all(SECTIONS.map(sectionData))};
